# Admin Web — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
> **Resume:** Check `PLAN.md` at project root to find current task. Mark tasks complete in both this file and PLAN.md.

**Goal:** Branch Staff and Owners can manage members, trainers, payments, and reminders via an Angular web app backed by new Spring Boot endpoints.

**Architecture:** New Flyway migrations add membership/payment/trainer tables. Spring Boot controllers enforce branch-scoping via a `BranchContext` helper that reads the authenticated user's branch from the DB. Angular app has a sidebar layout with role-aware nav (Staff sees their branch; Owner sees all). Each feature is a lazy-loaded Angular module.

**Tech Stack:** Spring Boot 3 (adds to Phase 0 backend), Angular 17 (standalone components inside modules), Angular Material for UI components, Chart.js via ng2-charts for revenue charts, ngx-csv-parser for import

**Prerequisite:** Phase 0 complete — `backend/`, `admin-web/` scaffolded and auth working.

```java
package in.gymculture.util;

import in.gymculture.enums.Role;
import in.gymculture.exception.UnauthorizedException;
import in.gymculture.model.User;
import in.gymculture.repository.UserRepository;
import org.springframework.security.core.Authentication;
import org.springframework.stereotype.Component;
import java.util.UUID;

@Component
public class BranchContext {

    private final UserRepository userRepository;

    public BranchContext(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    public User resolveUser(Authentication auth) {
        String firebaseUid = (String) auth.getPrincipal();
        return userRepository.findByFirebaseUid(firebaseUid)
                .orElseThrow(() -> new UnauthorizedException("User not registered"));
    }

    public UUID requireBranchId(Authentication auth) {
        User user = resolveUser(auth);
        if (user.getRole() == Role.OWNER) {
            throw new UnauthorizedException("Owners must specify a branch_id parameter");
        }
        return user.getBranch().getId();
    }

    public UUID resolveBranchId(Authentication auth, UUID requestedBranchId) {
        User user = resolveUser(auth);
        if (user.getRole() == Role.OWNER) {
            return requestedBranchId; // Owner can specify any branch
        }
        UUID userBranchId = user.getBranch().getId();
        if (requestedBranchId != null && !requestedBranchId.equals(userBranchId)) {
            throw new UnauthorizedException("Access to this branch is not permitted");
        }
        return userBranchId;
    }
}
```

```java
package in.gymculture.service;

import in.gymculture.dto.*;
import in.gymculture.exception.NotFoundException;
import in.gymculture.model.*;
import in.gymculture.repository.*;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.time.LocalDate;
import java.time.OffsetDateTime;
import java.util.*;

@Service
public class MembershipService {

    private final UserRepository userRepository;
    private final MembershipRepository membershipRepository;
    private final PtMembershipRepository ptMembershipRepository;
    private final PaymentLogRepository paymentLogRepository;

    public MembershipService(UserRepository userRepository,
                             MembershipRepository membershipRepository,
                             PtMembershipRepository ptMembershipRepository,
                             PaymentLogRepository paymentLogRepository) {
        this.userRepository = userRepository;
        this.membershipRepository = membershipRepository;
        this.ptMembershipRepository = ptMembershipRepository;
        this.paymentLogRepository = paymentLogRepository;
    }

    @Transactional
    public MembershipResponse activateMembership(ActivateMembershipRequest req, Branch branch) {
        User user = userRepository.findById(req.userId())
                .orElseThrow(() -> new NotFoundException("User not found: " + req.userId()));

        Membership membership = new Membership();
        membership.setUser(user);
        membership.setBranch(branch);
        membership.setStartDate(req.startDate());
        membership.setEndDate(req.endDate());
        membership.setActive(true);

        Membership saved = membershipRepository.save(membership);
        return new MembershipResponse(saved.getId(), user.getId(), user.getName(),
                saved.getStartDate(), saved.getEndDate(), saved.isActive());
    }

    @Transactional
    public PtMembership activatePtMembership(ActivatePtRequest req, Branch branch) {
        User user = userRepository.findById(req.userId())
                .orElseThrow(() -> new NotFoundException("User not found: " + req.userId()));
        User trainer = userRepository.findById(req.trainerId())
                .orElseThrow(() -> new NotFoundException("Trainer not found: " + req.trainerId()));

        PtMembership pt = new PtMembership();
        pt.setUser(user);
        pt.setBranch(branch);
        pt.setTrainer(trainer);
        pt.setStartDate(req.startDate());
        pt.setEndDate(req.endDate());
        pt.setTotalSessions(req.totalSessions());
        pt.setActive(true);
        pt.setActivatedAt(OffsetDateTime.now());

        return ptMembershipRepository.save(pt);
    }

    public void logPayment(LogPaymentRequest req, Branch branch, User recordedBy) {
        User user = userRepository.findById(req.userId())
                .orElseThrow(() -> new NotFoundException("User not found: " + req.userId()));

        PaymentLog log = new PaymentLog();
        log.setUser(user);
        log.setBranch(branch);
        log.setAmountPaise(req.amountPaise());
        log.setMethod(req.method());
        log.setPaymentDate(req.paymentDate());
        log.setNote(req.note());
        log.setRecordedBy(recordedBy);
```

```java
package in.gymculture.service;

import in.gymculture.dto.BranchStatsResponse;
import in.gymculture.repository.*;
import in.gymculture.model.Branch;
import org.springframework.stereotype.Service;

import java.time.LocalDate;
import java.util.List;
import java.util.UUID;

@Service
public class StatsService {

    private final BranchRepository branchRepository;
    private final MembershipRepository membershipRepository;
    private final PtMembershipRepository ptMembershipRepository;
    private final PaymentLogRepository paymentLogRepository;

    public StatsService(BranchRepository branchRepository,
                        MembershipRepository membershipRepository,
                        PtMembershipRepository ptMembershipRepository,
                        PaymentLogRepository paymentLogRepository) {
        this.branchRepository = branchRepository;
        this.membershipRepository = membershipRepository;
        this.ptMembershipRepository = ptMembershipRepository;
        this.paymentLogRepository = paymentLogRepository;
    }

    public BranchStatsResponse getBranchStats(UUID branchId) {
        Branch branch = branchRepository.findById(branchId)
                .orElseThrow(() -> new in.gymculture.exception.NotFoundException("Branch not found"));

        LocalDate now = LocalDate.now();
        LocalDate firstOfMonth = now.withDayOfMonth(1);

        long activeMembers = membershipRepository.countByBranchIdAndActiveTrue(branchId);
        long activePt = ptMembershipRepository.countByBranchIdAndActiveTrue(branchId);
        Long revenue = paymentLogRepository.sumAmountPaiseByBranchAndDateRange(branchId, firstOfMonth, now);
        long expiringPt = ptMembershipRepository
                .findByBranchIdAndEndDateBetween(branchId, now, now.plusDays(7)).size();

        return new BranchStatsResponse(
                branchId, branch.getName(), activeMembers, activePt,
                revenue != null ? revenue : 0L, expiringPt
        );
    }

    public List<BranchStatsResponse> getAllBranchStats() {
        return branchRepository.findAll().stream()
                .map(b -> getBranchStats(b.getId()))
                .toList();
    }
}
```

```ts
  constructor(private route: ActivatedRoute,
              private memberService: MemberService,
              private fb: FormBuilder) {
    this.membershipForm = fb.group({
      startDate: ['', Validators.required],
      endDate: ['', Validators.required],
    });
    this.ptForm = fb.group({
      trainerId: ['', Validators.required],
      startDate: ['', Validators.required],
      endDate: ['', Validators.required],
      totalSessions: [12, [Validators.required, Validators.min(1)]],
    });
    this.paymentForm = fb.group({
      amountRupees: [0, [Validators.required, Validators.min(1)]],
      method: ['CASH', Validators.required],
      paymentDate: [new Date().toISOString().split('T')[0], Validators.required],
      note: [''],
    });
  }

  ngOnInit() {
    const id = this.route.snapshot.paramMap.get('id')!;
    this.memberService.getMember(id).subscribe(m => this.member = m);
  }

  activateMembership() {
    if (!this.member || this.membershipForm.invalid) return;
    this.memberService.activateMembership({
      userId: this.member.id, ...this.membershipForm.value
    }).subscribe(() => this.successMsg = 'Membership activated.');
  }

  activatePt() {
    if (!this.member || this.ptForm.invalid) return;
    this.memberService.activatePt({
      userId: this.member.id, ...this.ptForm.value
    }).subscribe(() => this.successMsg = 'PT membership activated.');
  }

  logPayment() {
    if (!this.member || this.paymentForm.invalid) return;
    const v = this.paymentForm.value;
    this.memberService.logPayment({
      userId: this.member.id,
      amountPaise: Math.round(v.amountRupees * 100),
      method: v.method,
      paymentDate: v.paymentDate,
      note: v.note,
    }).subscribe(() => this.successMsg = 'Payment logged.');
  }
}
```

```ts
// import.module.ts
import { NgModule, Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterModule } from '@angular/router';
import { MatButtonModule } from '@angular/material/button';
import { HttpClient } from '@angular/common/http';
import { environment } from '../../../environments/environment';

@Component({
  selector: 'app-import',
  template: `
    <h2>Import Members</h2>
    <p>Upload a CSV or Excel file with columns: name, phone, email, membership_type,
       membership_start, membership_end, pt_start, pt_end, trainer_assigned</p>
    <input type="file" accept=".csv,.xlsx" (change)="onFileChange($event)" #fileInput>
    <button mat-raised-button color="primary" (click)="upload()" [disabled]="!file" style="margin-left:16px;">
      Upload
    </button>
    <p *ngIf="result" [style.color]="result.error ? 'red' : 'green'">{{ result.message }}</p>
  `
})
export class ImportComponent {
  file: File | null = null;
  result: { message: string; error: boolean } | null = null;

  constructor(private http: HttpClient) {}

  onFileChange(event: Event) {
    const input = event.target as HTMLInputElement;
    this.file = input.files?.[0] ?? null;
  }

  upload() {
    if (!this.file) return;
    const form = new FormData();
    form.append('file', this.file);
    this.http.post(`${environment.apiUrl}/import/members`, form).subscribe({
      next: (r: any) => this.result = { message: `Imported ${r.imported} members.`, error: false },
      error: (e) => this.result = { message: e.error?.error || 'Import failed', error: true }
    });
  }
}

@NgModule({
  declarations: [ImportComponent],
  imports: [CommonModule, MatButtonModule, RouterModule.forChild([{ path: '', component: ImportComponent }])]
})
export class ImportModule {}
```
